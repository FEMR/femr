@for(i <- 1 to genericTab.getLeftColumnSize) {

}

}

@defining(genericTab.getFields.get(i - 1)) { field => @dynamicField(i, field.getValue, field.getName, field.getSize, field.getOrder, field.getType, field.getPlaceholder, false) }
